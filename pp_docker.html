#5. Docker Compose

#Базовый compose.yaml

services:                                        # Описание всех сервисов приложения
  web:                                           # Сервис веб-приложения
    image: nginx:latest                          # Образ для запуска контейнера web
    ports:                                       # Список пробросов портов
      - "80:80"                                  # Порт 80 хоста -> порт 80 контейнера
    volumes:                                     # Монтирование томов/папок
      - ./html:/usr/share/nginx/html             # Локальная папка html -> папка в контейнере
    depends_on:                                  # Зависимости запуска
      - db                                       # Сначала должен быть поднят сервис db

  db:                                            # Сервис базы данных
    image: postgres:13                           # Образ PostgreSQL версии 13
    environment:                                 # Переменные окружения контейнера
      POSTGRES_PASSWORD: example                 # Пароль пользователя postgres
    volumes:                                     # Подключение тома для сохранения данных
      - db_data:/var/lib/postgresql/data         # Именованный том -> каталог данных БД

volumes:                                         # Объявление именованных томов
  db_data:                                       # Именованный том для данных PostgreSQL

#8. Разработка, очистка и безопасность

#Переменные окружения и ограничения ресурсов

КомандаОписание

#Советы по использованию Docker

#Рабочий процесс

  • Сначала проверяйте логи: docker logs -f CONTAINER.

  • Для сервисов из нескольких контейнеров используйте docker compose.

  • Фиксируйте версии образов тегами, не полагайтесь только на latest.

#Чистота и безопасность

  • Регулярно очищайте неиспользуемые ресурсы: docker system prune.

  • Не храните секреты прямо в Dockerfile или в открытых .env файлах.

  • Ограничивайте ресурсы контейнеров в dev и production.
