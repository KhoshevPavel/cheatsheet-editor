#Операторы сравнения

   
a == b a равно b
a != b a не равно b
a < b a меньше b
a > b a больше b
a <= b a меньше или равно b
a >= b a больше или равно b

#Операторы присваивания

Пример Эквивалентно
a += b То же, что a = a + b
a -= b То же, что a = a - b
a *= b То же, что a = a * b
a /= b То же, что a = a / b
a %= b То же, что a = a % b

#Логические операторы

Пример Значение
exp1 && exp2 Оба истинны (И)
exp1 || exp2 Хотя бы один истинен (ИЛИ)
!exp exp ложно (НЕ)

#Побитовые операторы

Оператор Описание
a & b Побитовое И
a | b Побитовое ИЛИ
a ^ b Побитовое исключающее ИЛИ
~ a Побитовое дополнение до единицы
a << b Побитовый сдвиг влево
a >> b Побитовый сдвиг вправо

#Тернарный оператор

           ┌── Истина ──┐
Result = Condition ? Exp1 : Exp2;
           └───── Ложь ─────┘

int x = 3, y = 5, max;
max = (x > y) ? x : y;

// Вывод: 5
std::cout << max << std::endl;

int x = 3, y = 5, max;
if (x > y) {
    max = x;
} else {
    max = y;
}
// Вывод: 5
std::cout << max << std::endl;

#for_each (начиная с C++11)

#include <iostream>
#include <array> int main()
{
    auto print = [](int num) { std::cout << num << std::endl; };

    std::array<int, 4> arr = {1, 2, 3, 4};
    std::for_each(arr.begin(), arr.end(), print);
    return 0;
}

#Классы и объекты C++

#Определение класса

class MyClass {
  public:             // модификатор доступа 
    int myNum;        // атрибут (int)
    string myString;  // атрибут (string)
};

#Создание объекта

MyClass myObj;  // Создать объект MyClass

myObj.myNum = 15;          // Установить myNum в 15
myObj.myString = "Hello";  // Установить myString в "Hello"

cout << myObj.myNum << endl;         // Вывод 15
cout << myObj.myString << endl;      // Вывод "Hello"

#Конструкторы

class MyClass {
  public:
    int myNum;
    string myString;
    MyClass() {  // конструктор
      myNum = 0;
      myString = "";
    }
};

MyClass myObj;  // Создать объект MyClass

cout << myObj.myNum << endl;         // Вывод 0
cout << myObj.myString << endl;      // Вывод ""

#Деструкторы

class MyClass {
  public:
    int myNum;
    string myString;
    MyClass() {  // конструктор
      myNum = 0;
      myString = "";
    }
    ~MyClass() {  // деструктор
      cout << "Object destroyed." << endl;
    }
};

MyClass myObj;  // Создать объект MyClass 
// Код здесь... 
// Объект уничтожается при выходе из области видимости

#Методы класса

class MyClass {
  public:
    int myNum;
    string myString;
    void myMethod() {  // метод, определённый в классе
      cout << "Hello World!" << endl;
    }
};

MyClass myObj;  // Создать объект MyClass
myObj.myMethod();  // вызов метода

#Модификаторы доступа

class MyClass {
  public:     // публичный доступ 
    int x;    // публичный атрибут 
  private:    // приватный доступ 
    int y;    // приватный атрибут 
  protected:  // защищённый доступ 
    int z;    // защищённый атрибут
};

MyClass myObj;
myObj.x = 25;  // разрешено (public)
myObj.y = 50;  // запрещено (private)
myObj.z = 75;  // запрещено (protected)

#Геттеры и сеттеры

class MyClass {
  private:
    int myNum;
  public:
    void setMyNum(int num) {  // сеттер
      myNum = num;
    }
    int getMyNum() {  // геттер return myNum;
    }
};

MyClass myObj;
myObj.setMyNum(15);  // Установить myNum в 15
cout << myObj.getMyNum() << endl;  // Вывод 15

#Наследование

class Vehicle {
  public:
    string brand = "Ford";
    void honk() {
      cout << "Tuut, tuut!" << endl;
    }
};

class Car : public Vehicle {
  public:
    string model = "Mustang";
};

Car myCar;
myCar.honk();  // Вывод "Tuut, tuut!"
cout << myCar.brand + " " + myCar.model << endl;  // Вывод "Ford Mustang"
