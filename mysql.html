MySQL cheatsheet

Шпаргалка по SQL-командам для работы с MySQL.

#Начало работы

#Подключить MySQL

mysql -u <user> -p

mysql [db_name]

mysql -h <host> -P <port> -u <user> -p [db_name]

mysql -h <host> -u <user> -p [db_name]

#Общие команды

#Database

- -
CREATE DATABASE db ; Создать БД
SHOW DATABASES; Список БД
USE db; Переключиться на БД
CONNECT db ; Переключиться на БД
DROP DATABASE db; Удалить БД

#Таблица

- -
SHOW TABLES; Список таблиц текущей БД
SHOW FIELDS FROM t; Список полей таблицы
DESC t; Показать структуру таблицы
SHOW CREATE TABLE t; Показать SQL создания таблицы
TRUNCATE TABLE t; Удалить все данные из таблицы
DROP TABLE t; Удалить таблицу

#Process

- -
show processlist; Список процессов
kill pid; Завершить процесс

#Прочее

- -
exit or \q Выход из сессии MySQL

#Резервные копии

Создать резервную копию

mysqldump -u user -p db_name > db.sql

Экспорт БД без схемы

mysqldump -u user -p db_name --no-data=true --add-drop-table=false > db.sql

Восстановить резервную копию

mysql -u user -p db_name < db.sql

#Примеры MySQL

#Управление таблицами

Создать новую таблицу с тремя столбцами

CREATE TABLE t (
     id    INT,
     name  VARCHAR DEFAULT NOT NULL,
     price INT DEFAULT 0 PRIMARY KEY(id)
);

Удалить таблицу из базы данных

DROP TABLE t ;

Добавить новый столбец в таблицу

ALTER TABLE t ADD column;

Удалить столбец c из таблицы

ALTER TABLE t DROP COLUMN c ;

Добавить ограничение

ALTER TABLE t ADD constraint;

Удалить ограничение

ALTER TABLE t DROP constraint;

Переименовать таблицу из t1 в t2

ALTER TABLE t1 RENAME TO t2;

Переименовать столбец c1 в c2

ALTER TABLE t1 RENAME c1 TO c2 ;

Удалить все данные из таблицы

TRUNCATE TABLE t;

#Запрос данных из таблицы

Выбрать данные из столбцов c1, c2 таблицы

SELECT c1, c2 FROM t

Выбрать все строки и столбцы из таблицы

SELECT * FROM t

Выбрать данные и отфильтровать строки по условию

SELECT c1, c2 FROM t
WHERE condition

Выбрать уникальные строки из таблицы

SELECT DISTINCT c1 FROM t
WHERE condition

Сортировать результат по возрастанию или убыванию

SELECT c1, c2 FROM t
ORDER BY c1 ASC [DESC]

Пропустить offset строк и вернуть следующие n строк

SELECT c1, c2 FROM t
ORDER BY c1
LIMIT n OFFSET offset

Группировать строки с помощью агрегатной функции

SELECT c1, aggregate(c2)
FROM t
GROUP BY c1

Фильтровать группы с помощью HAVING

SELECT c1, aggregate(c2)
FROM t
GROUP BY c1
HAVING condition

#Запрос из нескольких таблиц

Внутреннее соединение t1 и t2

SELECT c1, c2
FROM t1
INNER JOIN t2 ON condition

Левое соединение t1 и t2

SELECT c1, c2
FROM t1
LEFT JOIN t2 ON condition

Правое соединение t1 и t2

SELECT c1, c2
FROM t1
RIGHT JOIN t2 ON condition

Полное внешнее соединение

SELECT c1, c2
FROM t1
FULL OUTER JOIN t2 ON condition

Декартово произведение строк таблиц

SELECT c1, c2
FROM t1
CROSS JOIN t2

Другой способ перекрёстного соединения

SELECT c1, c2
FROM t1, t2

Соединение t1 с собой через INNER JOIN

SELECT c1, c2
FROM t1 A
INNER JOIN t1 B ON condition

Using SQL Operators Combine rows from two queries

SELECT c1, c2 FROM t1
UNION [ALL]
SELECT c1, c2 FROM t2

Пересечение результатов двух запросов

SELECT c1, c2 FROM t1
INTERSECT
SELECT c1, c2 FROM t2

Вычесть один результат из другого

SELECT c1, c2 FROM t1
MINUS
SELECT c1, c2 FROM t2

Поиск строк по шаблону %, _

SELECT c1, c2 FROM t1
WHERE c1 [NOT] LIKE pattern

Поиск строк в списке

SELECT c1, c2 FROM t
WHERE c1 [NOT] IN value_list

Поиск строк между двумя значениями

SELECT c1, c2 FROM t
WHERE  c1 BETWEEN low AND high

Проверка на NULL в таблице

SELECT c1, c2 FROM t
WHERE  c1 IS [NOT] NULL

#Использование ограничений SQL

Установить c1 и c2 как первичный ключ

CREATE TABLE t(
    c1 INT, c2 INT, c3 VARCHAR,
    PRIMARY KEY (c1,c2)
);

Установить c2 как внешний ключ

CREATE TABLE t1(
    c1 INT PRIMARY KEY,
    c2 INT,
    FOREIGN KEY (c2) REFERENCES t2(c2)
);

Сделать значения в c1 и c2 уникальными

CREATE TABLE t(
    c1 INT, c1 INT,
    UNIQUE(c2,c3)
);

Обеспечить c1 > 0 и c1 >= c2

CREATE TABLE t(
  c1 INT, c2 INT,
  CHECK(c1> 0 AND c1 >= c2)
);

Запретить NULL в c2

CREATE TABLE t(
     c1 INT PRIMARY KEY,
     c2 VARCHAR NOT NULL
);

#Изменение данных

Вставить одну строку в таблицу

INSERT INTO t(column_list)
VALUES(value_list);

Вставить несколько строк в таблицу

INSERT INTO t(column_list)
VALUES (value_list),
       (value_list), …;

Вставить строки из t2 в t1

INSERT INTO t1(column_list)
SELECT column_list
FROM t2;

Обновить c1 для всех строк

UPDATE t
SET c1 = new_value;

Обновить c1, c2 по условию

UPDATE t
SET c1 = new_value,
        c2 = new_value
WHERE condition;

Удалить все данные таблицы

DELETE FROM t;

Удалить часть строк таблицы

DELETE FROM t
WHERE condition;

#Управление представлениями

Создать представление из c1 и c2

CREATE VIEW v(c1,c2)
AS
SELECT c1, c2
FROM t;

Создать представление с check option

CREATE VIEW v(c1,c2)
AS
SELECT c1, c2
FROM t;
WITH [CASCADED | LOCAL] CHECK OPTION;

Создать рекурсивное представление

CREATE RECURSIVE VIEW v
AS
select-statement -- anchor part
UNION [ALL]
select-statement; -- recursive part

Создать временное представление

CREATE TEMPORARY VIEW v
AS
SELECT c1, c2
FROM t;

Удалить представление

DROP VIEW view_name;

#Управление триггерами

Создать или изменить триггер

CREATE OR MODIFY TRIGGER trigger_name
WHEN EVENT
ON table_name TRIGGER_TYPE
EXECUTE stored_procedure;

#WHEN

- -
BEFORE invoke before the event occurs
AFTER invoke after the event occurs

#EVENT

- -
INSERT invoke for INSERT
UPDATE invoke for UPDATE
DELETE invoke for DELETE

#TRIGGER_TYPE

- -
FOR EACH ROW
FOR EACH STATEMENT

#Управление индексами

Создать индекс по c1 и c2 таблицы t

CREATE INDEX idx_name
ON t(c1,c2);

Создать уникальный индекс по c3, c4 таблицы t

CREATE UNIQUE INDEX idx_name
ON t(c3,c4)

Удалить индекс

DROP INDEX idx_name ON t;
