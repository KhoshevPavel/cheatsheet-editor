#Строки

char greetings[] = "Hello World!";
printf("%s", greetings);

Доступ к строке

char greetings[] = "Hello World!";
printf("%c", greetings[0]);

Изменить строку

char greetings[] = "Hello World!";
greetings[0] = 'J';

printf("%s", greetings);
// выводит "Jello World!"

Другой способ создать строку

char greetings[] = {'H','e','l','l','\0'};

printf("%s", greetings);
// вывести "Hell!"

Создание строки через указатель на символ (строковые литералы)

char *greetings = "Hello";
printf("%s", greetings);
// вывести "Hello!"

NOTE: String literals might be stored in read-only section of memory. Modifying a string literal invokes undefined behavior. You can't modify it!

В C нет типа String, используйте тип char и создайте массив символов

int i = 0;

  i++;

NOTE: Don't forget to increment the variable used in the condition, otherwise the loop will never end and become an "infinite loop"!

#Адрес в памяти

При создании переменной ей назначается адрес в памяти

int myAge = 43;

printf("%p", &myAge);
// Вывод: 0x7ffe5367e044

To access it, use the reference operator (&)

#Создание указателя

int myAge = 43; // переменная типа int
printf("%d\n", myAge); // вывести значение myAge (43) // Вывести адрес памяти переменной myAge (0x7ffe5367e044)
printf("%p\n", &myAge);

#Переменная-указатель

int myAge = 43; // переменная типа int
int*ptr = &myAge; // переменная-указатель с именем ptr, используется для хранения адреса myAge printf("%d\n", myAge); // вывести значение myAge (43) printf("%p\n", &myAge); // вывести адрес памяти myAge (0x7ffe5367e044)
printf("%p\n", ptr); // использовать указатель (0x7ffe5367e044), чтобы вывести адрес памяти myAge

#Разыменование

int myAge = 43; // объявление переменной
int*ptr = &myAge; // объявление указателя // Ссылка: вывести myAge через указатель
// адрес памяти (0x7ffe5367e044)
printf("%p\n", ptr);
// разыменование: вывести значение myAge через указатель (43)
printf("%d\n", *ptr);

#Предопределенные макросы

Макрос Описание
__DATE__ Текущая дата, символьная константа в формате "MMM DD YYYY"

#Пример предопределённого макроса

#include <stdio.h> int main(void) {

#Оператор продолжения макроса (\)

Макрос обычно записывается в одну строку.

#define message_for(a, b) \
    printf(#a " and " #b ": We love you!\n")

Если макрос слишком длинный для одной строки, используйте оператор продолжения макроса \

#Оператор # (стрингификация)

#include <stdio.h> #define message_for(a, b) \
  printf(#a " and " #b ": We love you!\n") int main(void) {
  message_for(Carole, Debra);

  return 0;
}

При компиляции и запуске этого кода вы получите следующий результат:

Carole and Debra: We love you!

Чтобы преобразовать параметр макроса в строковую константу, используйте оператор стрингификации #

#Оператор ## (конкатенация токенов)

#include <stdio.h> #define tokenpaster(n) printf ("Token " #n " = %d\n", token##n) int main(void) {
  int token34 = 40;
  tokenpaster(34);

  return 0;
}

ab Открыть двоичный файл в режиме добавления

#Открытие файла: fopen()

#include <stdio.h> void main() {
  FILE *fp;
  char ch;

  fp = fopen("file_handle.c", "r");

  while (1) {
    ch = fgetc(fp);
    if (ch == EOF)
      break;
    printf("%c", ch);
  }
  fclose(fp);
}

Установить файловый указатель в заданную позицию

#rewind()

#include <stdio.h>
#include <conio.h> void main() {
  FILE *fp;
  char c;

  clrscr();

  fp = fopen("file.txt", "r");

  while( (c = fgetc(fp) ) != EOF) {
    printf("%c", c);
  }

  rewind(fp); // переместить файловый указатель в начало файла while( (c = fgetc(fp) ) != EOF) {
    printf("%c", c);
  }
  fclose(fp);

  getch();
}
// вывод
// Привет, мир! Привет, мир!

#ftell()

#include <stdio.h>
#include <conio.h> void main () {
   FILE *fp;
   int length;

   clrscr();

   fp = fopen("file.txt", "r");

   fseek(fp, 0, SEEK_END);
   length = ftell(fp); // вернуть текущую позицию
   fclose(fp);

   printf("File size: %d bytes", length);

   getch();
}
// вывод
// размер файла: 18 байт
