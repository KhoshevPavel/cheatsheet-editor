Curl cheatsheet

Шпаргалка по curl с командами и примерами основных приёмов.

#Начало работы

#Введение

Curl — инструмент для передачи данных между серверами, поддерживает протоколы:

  • HTTP
  • HTTPS
  • FTP
  • IMAP
  • LDAP
  • POP3
  • SCP
  • SFTP
  • SMB
  • SMTP
  • etc...
  • Репозиторий исходников curl на GitHub (github.com)
  • Официальный сайт curl (curl.se)

#Параметры

-o <file>    # записать в файл
-u user:pass # аутентификация

-v   # --verbose: подробный вывод при выполнении
-vv  # более подробный вывод
-s   # --silent: не показывать прогресс и ошибки
-S   # --show-error: вместе с -sS показывать только ошибки

-i  # включить HTTP-заголовки в вывод
-I  # только заголовок

#Запрос

-X POST # --request
-L # Если страница перенаправляет, следовать по ссылке
-F # данные HTTP POST для multipart/form-data

#Данные (--data)

# данные HTTP POST
# URL-кодирование (напр. status="Hello")
-d 'data' # передать файл с данными
-d @file

# отправить данные -d через GET
-G

#Информация в заголовке

-A <str>      # --user-agent

-b name=val   # --cookie

-b, --cookie FILE           # Загрузить cookies из файла для URL
-c, --cookie-jar FILE       # Сохранить cookies в файл из URL

-H "X-Foo: y" # --header

--compressed  # использовать deflate/gzip

#SSL

    --cacert <file>
    --capath <dir>

-E, --cert <cert> # сертификат клиента
    --cert-type # der/pem/eng
-k, --insecure # Для самоподписанных сертификатов

#Установка

apk add --update curl # установить в Alpine Linux

#Пример

#CURL GET/HEAD

команда описание
curl -I https:// curl отправляет запрос
curl -v -I https:// curl запрос с подробным выводом
curl -X GET https:// явный HTTP-метод для curl
curl --noproxy 127.0.0.1 http://www.stackoverflow.com curl без HTTP-прокси
curl --connect-timeout 10 -I -k https:// curl без таймаута по умолчанию
curl --verbose --header "Host: www.mytest.com:8182" curl с дополнительным заголовком
curl -k -v https://www.google.com curl получить ответ с заголовками

#Загрузка нескольких файлов

$ curl -v --include \
--form key1=value1 \
    --form upload=@localfilename URL

#Форматирование JSON-ответа curl

$ curl -XGET http://${elasticsearch_ip}:9200/_cluster/nodes | python -m json.tool

#CURL POST

команда описание
curl -d "name=username&password=123456" <URL> curl отправить запрос
curl <URL> -H "content-type: application/json" -d "{ \"woof\": \"bark\"}" curl отправляет JSON

#Скрипт установки rvm через CURL

curl -sSL https://get.rvm.io | bash

#Расширенные примеры CURL

команда описание
curl -L -s http://ipecho.net/plain, curl -L -s http://whatismijnip.nl получить мой публичный IP
curl -u $username:$password http://repo.dennyzhang.com/README.txt curl с учётными данными
curl -v -F key1=value1 -F upload=@localfilename <URL> загрузка через curl
curl -k -v --http2 https://www.google.com/ использовать HTTP/2 в curl
curl -T cryptopp552.zip -u test:test ftp://10.32.99.187/ загрузка через curl по FTP
curl -u test:test ftp://10.32.99.187/cryptopp552.zip -o cryptopp552.zip скачивание через curl по FTP
curl -v -u admin:admin123 --upload-file package1.zip http://mysever:8081/dir/package1.zip загрузка с учётными данными через curl

#Проверка времени ответа сайта

curl -s -w \
'\nLookup time:\t%{time_namelookup}\nConnect time:\t%{time_connect}\nAppCon time:\t%{time_appconnect}\nRedirect time:\t%{time_redirect}\nPreXfer time:\t%{time_pretransfer }\nStartXfer time:\t%{time_starttransfer}\n\nTotal time:\t%{time_total}\n' \
     -o /dev/null https://www.google.com

#Проверка доступности удалённого ресурса

curl -o /dev/null --silent -Iw "%{http_code}" https://example.com/my.remote.tarball.gz

#Загрузка файла

curl https://example.com | \
grep --only-matching 'src="[^"]*.[png]"' | \
cut -d \" -f2 | \
while read i; do curl https://example.com/"${i}" \
-o "${i##*/}"; done

Скачать все PNG-файлы с сайта (с помощью GNU grep)

#Сохранение файла без изменения имени

curl --remote-name "https://example.com/linux-distro.iso"

Переименовать файл

curl --remote-name "http://example.com/index.html" --output foo.html

#Продолжение прерванной загрузки

curl --remote-name --continue-at -"https://example.com/linux-distro.iso"

#Загрузка с нескольких доменов

curl "https://www.{example,w3,iana}.org/index.html" --output "file_#1.html"

#Скачивание серии файлов

curl "https://{foo,bar}.com/file_[1-4].webp" --output "#1_#2.webp"

Скачать серию файлов (результат: foo_file1.webp, foo_file2.webp...bar_file1_webp и т.д.)

#Перенаправление вывода в файл

$ curl http://url/file > file

#Базовая аутентификация

$ curl --user username:password http://example.com/
$ curl -u username:password http://example.com/

#Запись в файл (вместо stdout)

$ curl -o file http://url/file
$ curl --output file http://url/file

#Загрузка информации из заголовков

$ curl -I url
# показать заголовки

#Запись вывода в файл Remote_file

$ curl -o file http://url/file
$ curl --output file http://url/file

#Выполнение удалённого скрипта

$ curl -s http://url/myscript.sh

#Конфигурационный файл

curl -K file
# прочитать конфигурацию из файла
curl --config file
$HOME/.curlrc # файл конфигурации по умолчанию в UNIX-системах
