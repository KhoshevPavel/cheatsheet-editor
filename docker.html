#Начало работы

#Начало работы

Создание и запуск контейнера в фоне

$ docker run -d -p 80:80 docker/getting-started
  • -d — запуск контейнера в отсоединённом режиме
  • -p 80:80 — проброс порта 80 в порт 80 контейнера
  • docker/getting-started — используемый образ

Создание и запуск контейнера на переднем плане

$ docker run -it -p 8001:8080 --name my-nginx nginx
  • -it — интерактивный режим bash
  • -p 8001:8080 — проброс порта 8001 в порт 8080 контейнера
  • --name my-nginx — указание имени
  • nginx — используемый образ

#Создание

docker create [options] IMAGE
-a, --attach              # подключить stdout/err
-i, --interactive         # подключить stdin (интерактивно)
-t, --tty                 # pseudo-tty
--name NAME               # назвать контейнер
-p, --publish 5000:5000    # маппинг портов (хост:контейнер)
--expose 5432             # открыть порт для контейнеров
-P, --publish-all         # опубликовать все порты
--link container:alias    # линковка
-v, --volume `pwd`:/app   # монтирование (нужны абсолютные пути)
-e, --env NAME=hello      # переменные окружения

#Отладка и инспекция

#Inspect + format

$ docker inspect my-nginx
$ docker inspect -f '{{.State.Status}}' my-nginx
$ docker inspect -f '{{range .NetworkSettings.Networks}}{{.IPAddress}}{{end}}' my-nginx
$ docker ps --format 'table {{.Names}}\\t{{.Status}}\\t{{.Ports}}'
  • inspect — полный JSON c метаданными контейнера
  • -f '{{...}}' — вывести только нужное поле через Go template
  • .State.Status — быстро проверить текущий статус контейнера
  • .NetworkSettings... — получить IP контейнера в сети
  • ps --format — кастомный вывод колонок для быстрого обзора

#Логи и события

$ docker logs -f --tail=200 my-nginx
$ docker logs --since=30m my-nginx
$ docker events --since 1h
$ docker stats --no-stream
  • -f — следить за логом в реальном времени
  • --tail=200 — показать последние 200 строк перед follow
  • --since=30m — логи только за последние 30 минут
  • events — события Docker daemon (start, die, pull и т.д.)
  • stats --no-stream — одноразовый снимок ресурсов без постоянного потока

#Копирование и shell-debug

$ docker cp my-nginx:/etc/nginx/nginx.conf .
$ docker cp ./index.html my-nginx:/usr/share/nginx/html/
$ docker exec -it my-nginx sh
$ docker exec -it --user root my-nginx sh
  • docker cp container:path . — копировать файл из контейнера на хост
  • docker cp ./file container:path — копировать файл с хоста в контейнер
  • exec -it ... sh — интерактивная shell-сессия внутри контейнера
  • --user root — запуск команды от root (для диагностики прав/файлов)

#Тома и монтирование

#Управление томами

$ docker volume create app_data
$ docker volume inspect app_data
$ docker volume ls --filter dangling=true
$ docker volume rm app_data
  • volume create — создать именованный volume
  • volume inspect — посмотреть mountpoint и метаданные тома
  • --filter dangling=true — тома, не используемые контейнерами
  • volume rm — удалить том (только если не занят контейнером)

#bind mount vs named volume

$ docker run -d --name app -v app_data:/var/lib/app my-app
$ docker run -d --name app -v /abs/path:/app:ro my-app
$ docker run -d --mount type=bind,src=/abs/path,dst=/app,readonly my-app
  • -v app_data:/var/lib/app — named volume, управляется Docker
  • -v /abs/path:/app:ro — bind mount из директории хоста в контейнер read-only
  • --mount ... — более явный и безопасный синтаксис для сложных mount-конфигураций

Для хоста всегда используйте абсолютные пути.

#Безопасность и лимиты

#Hardening runtime

$ docker run --read-only --tmpfs /tmp my-app
$ docker run --cap-drop ALL --cap-add NET_BIND_SERVICE my-app
$ docker run --user 1000:1000 my-app
$ docker run --security-opt no-new-privileges my-app
  • --read-only — корневая ФС контейнера только для чтения
  • --tmpfs /tmp — временная writable-область в памяти
  • --cap-drop/--cap-add — минимально необходимые Linux capabilities
  • --user — запуск без root, по UID:GID
  • no-new-privileges — запрет эскалации прав внутри процесса

#CPU / RAM / restart policy

$ docker run -m 512m --memory-swap 512m my-app
$ docker run --cpus=1.5 --pids-limit=200 my-app
$ docker run --restart unless-stopped my-app
  • -m — лимит RAM контейнера
  • --memory-swap 512m — ограничить swap (равно RAM: swap фактически отключён)
  • --cpus — ограничение CPU-квоты
  • --pids-limit — ограничение количества процессов
  • --restart unless-stopped — автоподнятие после перезапуска хоста/демона

#Buildx и multi-platform

#Сборка для нескольких архитектур

$ docker buildx create --use --name mybuilder
$ docker buildx ls
$ docker buildx build --platform linux/amd64,linux/arm64 -t myorg/my-app:latest --push .
  • buildx create --use — создать и активировать builder
  • --platform — собрать multi-arch образ
  • --push — сразу отправить multi-arch manifest в registry

#Кэширование сборки

$ docker buildx build --cache-from type=registry,ref=myorg/my-app:cache \
>   --cache-to type=registry,ref=myorg/my-app:cache,mode=max \
>   -t myorg/my-app:latest --push .
  • --cache-from — использовать удалённый кэш предыдущих сборок
  • --cache-to ... mode=max — выгрузить максимум кэша для следующих CI-джоб
  • Сильно ускоряет повторные build в CI/CD
